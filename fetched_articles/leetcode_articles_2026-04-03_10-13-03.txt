m=len(grid)
n=len(grid[0])
def helper(r,c):
if r==m-1 and c==n-1:
return grid[m-1][n-1]
if r>=m or c>=n:
return float('inf')
right = helper(r,c+1)
down = helper(r+1,c)
return min(right,down) + grid[r][c]
return helper(0,0)
